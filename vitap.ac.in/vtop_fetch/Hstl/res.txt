  (() => {
    const dropdownIds = ["hostelBlock", "room_type", "floor", "rommatecount"];
    const originalSnapshots = {};
    const legitimateSnapshots = {}; // Track legitimate states separately
    const activeObservers = {};
    
    // Define dependency relationships
    const dependencyChain = {
        "hostelBlock": ["room_type", "floor", "rommatecount"], // hostelBlock affects all others
        "room_type": ["floor", "rommatecount"], // room_type affects floor and rommatecount
        "floor": ["rommatecount"] // floor affects rommatecount
    };
    
    // Track legitimate updates to avoid false tampering alerts
    const legitimateUpdates = new Set();
    const pendingLegitimateUpdates = new Map(); // Track timing of legitimate updates
    
    //console.log('Dropdown Monitor Script loaded with dependency handling');
    
    // Save current dropdown options and set up MutationObserver
    function storeOriginalAndObserve(id) {
        console.log(`Attempting to observe dropdown: ${id}`);
        
        const el = document.getElementById(id);
        if (!el) {
            console.error(`Element with id '${id}' not found to observe.`);
            return false;
        }
        
        // Check if element is a select element
        if (el.tagName.toLowerCase() !== 'select') {
            console.error(`Element with id '${id}' is not a select element.`);
            return false;
        }
        
        // Store original snapshot (never changes)
        originalSnapshots[id] = Array.from(el.options).map(opt => ({
            value: opt.value,
            text: opt.text.trim()
        }));
        
        // Store legitimate snapshot (updates with legitimate changes)
        legitimateSnapshots[id] = [...originalSnapshots[id]];
        
        console.log(`Stored snapshots for dropdown: ${id} with ${el.options.length} options.`);
        
        // Disconnect previous observer if exists
        if (activeObservers[id]) {
            activeObservers[id].disconnect();
            console.log(`Disconnected previous observer for ${id}`);
        }
        
        // Create a new observer for this dropdown
        const observer = new MutationObserver(mutations => {
            console.log(`Mutation detected in dropdown: ${id}`);
            
            // Small delay to allow DOM to stabilize
            setTimeout(() => {
                checkForTampering(id);
            }, 50);
        });
        
        // Observe changes to child elements (option additions/removals)
        observer.observe(el, {
            childList: true,
            subtree: true,
            attributes: true,
            attributeFilter: ['value', 'selected']
        });
        
        activeObservers[id] = observer;
        console.log(`Successfully set up observer for dropdown: ${id}`);
        return true;
    }
    
    // Main tampering check function
    function checkForTampering(id) {
        const el = document.getElementById(id);
        if (!el) return;
        
        const current = Array.from(el.options).map(opt => ({
            value: opt.value,
            text: opt.text.trim()
        }));
        
        const isLegitimateUpdate = legitimateUpdates.has(id);
        const hasChangedFromLegitimate = JSON.stringify(current) !== JSON.stringify(legitimateSnapshots[id]);
        
        if (hasChangedFromLegitimate) {
            if (isLegitimateUpdate) {
                // This is a legitimate change - update the legitimate snapshot
                console.log(`Legitimate change detected for ${id}, updating legitimate snapshot`);
                legitimateSnapshots[id] = [...current];
                legitimateUpdates.delete(id);
                
                // Clear any pending timeout for this dropdown
                if (pendingLegitimateUpdates.has(id)) {
                    clearTimeout(pendingLegitimateUpdates.get(id));
                    pendingLegitimateUpdates.delete(id);
                }
            } else {
                // This is tampering - compare against legitimate snapshot
                console.log(`Potential tampering detected for ${id}`);
                onTampering(id);
            }
        }
    }
    
    // Mark dependent dropdowns as having legitimate updates
    function markDependentUpdates(parentId) {
        if (dependencyChain[parentId]) {
            dependencyChain[parentId].forEach(dependentId => {
                legitimateUpdates.add(dependentId);
                console.log(`Marked ${dependentId} as legitimate update due to ${parentId} change`);
                
                // Set a timeout to clear the legitimate update flag if no change occurs
                if (pendingLegitimateUpdates.has(dependentId)) {
                    clearTimeout(pendingLegitimateUpdates.get(dependentId));
                }
                
                const timeoutId = setTimeout(() => {
                    if (legitimateUpdates.has(dependentId)) {
                        legitimateUpdates.delete(dependentId);
                        console.log(`Cleared pending legitimate update for ${dependentId} due to timeout`);
                    }
                    pendingLegitimateUpdates.delete(dependentId);
                }, 2000); // 2 second timeout
                
                pendingLegitimateUpdates.set(dependentId, timeoutId);
            });
        }
    }
    
    // Action on tampering detection
    function onTampering(id) {
        const message = `⚠️ Dropdown tampering detected in '${id}'. This may be logged.`;
        violaterulehostelBlock(message);
        
        // Log detailed tampering information
        logTamperingDetails(id);
        
        // Optional: call your violation logging function here
        // violateruleroomType();
        
        // Optional: restore legitimate options

    }
    
    // Log detailed tampering information

        const el = document.getElementById(id);
        if (!el) return;
        
        const current = Array.from(el.options).map(opt => ({
            value: opt.value,
            text: opt.text.trim()

        console.log('Current count:', current.length); */
        
        // Compare against legitimate state
        const addedOptions = current.filter(curr => 
            !legitimate.some(leg => leg.value === curr.value && leg.text === curr.text)
        );
        
        const removedOptions = legitimate.filter(leg => 
            !current.some(curr => curr.value === leg.value && curr.text === leg.text)
        );
        
        if (addedOptions.length > 0) {
            //console.log('⚠️ Unauthorized options added:', addedOptions);
        }
        if (removedOptions.length > 0) {
         //   console.log('⚠️ Legitimate options removed:', removedOptions);
        }
        
        // Log tampering type
        if (current.length > legitimate.length) {

        } else if (current.length < legitimate.length) {
           // console.log('⚠️ Tampering type: Options removed');
        } else {
            //console.log('⚠️ Tampering type: Options modified');
        }
    }
    
    // Set up change listeners for parent dropdowns
    function setupDependencyListeners() {
        Object.keys(dependencyChain).forEach(parentId => {
            const parentEl = document.getElementById(parentId);
            if (parentEl) {
                parentEl.addEventListener('change', function() {
                    console.log(`Parent dropdown ${parentId} changed, marking dependents as legitimate`);
                    markDependentUpdates(parentId);
                });
                console.log(`Set up dependency listener for ${parentId}`);
            }
        });
    }
    
    // Restore legitimate options
    function restoreLegitimateOptions(id) {
        const el = document.getElementById(id);
        if (!el || !legitimateSnapshots[id]) return;
        
        // Clear existing options
        el.innerHTML = '';
        
        // Restore legitimate options
        legitimateSnapshots[id].forEach(optData => {
            const option = document.createElement('option');
            option.value = optData.value;
            option.text = optData.text;
            el.appendChild(option);
        });
        
        console.log(`Restored legitimate options for ${id}`);
    }
    
    // Initialize observers for existing dropdowns
    function initializeObservers() {
      //  console.log('Initializing observers...');
       // console.log('Document ready state:', document.readyState);
        
        let foundElements = 0;
        
        dropdownIds.forEach(id => {
            const el = document.getElementById(id);
            console.log(`Checking element ${id}:`, el ? 'Found' : 'Not found');
            
            if (el) {
                foundElements++;
                if (storeOriginalAndObserve(id)) {
                    console.log(`Successfully initialized observer for ${id}`);
                }
            }
        });
        
        console.log(`Found ${foundElements} out of ${dropdownIds.length} expected dropdowns`);
        
        // Set up dependency listeners
        setupDependencyListeners();
        
        // If not all elements found, set up document observer
        if (foundElements < dropdownIds.length) {
            //console.log('Setting up document observer for missing elements...');
            setupDocumentObserver();
        }
    }
    
    // Set up observer for dynamically added dropdowns
    function setupDocumentObserver() {
        const documentObserver = new MutationObserver(() => {
            dropdownIds.forEach(id => {
                const el = document.getElementById(id);
                if (el && !activeObservers[id]) {
                    console.log(`Found previously missing element: ${id}`);
                    storeOriginalAndObserve(id);
                    setupDependencyListeners(); // Re-setup listeners
                }
            });
        });
        
        documentObserver.observe(document.body, {
            childList: true,
            subtree: true
        });
        
       // console.log('Document observer set up for dynamic elements');
    }
    
    // Multiple initialization strategies
    function initialize() {
        //console.log('Starting initialization...');
        
        // Strategy 1: Immediate initialization
        initializeObservers();
        
        // Strategy 2: DOM Content Loaded
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', () => {
               // console.log('DOM Content Loaded - reinitializing...');
                setTimeout(initializeObservers, 100);
            });
        }
        
        // Strategy 3: Window Load (for slower loading content)
        window.addEventListener('load', () => {
           // console.log('Window loaded - final initialization...');
            setTimeout(initializeObservers, 500);
        });
    }
    
    // Expose functions globally
    window.storeOriginalAndObserve = storeOriginalAndObserve;
    window.restoreLegitimateOptions = restoreLegitimateOptions;
    window.initializeDropdownObservers = initializeObservers;
    window.markDependentUpdates = markDependentUpdates;
    
    // Debug function
    window.debugDropdowns = function() {
        /* console.log('=== Dropdown Debug Info ===');
        console.log('Expected dropdowns:', dropdownIds);
        console.log('Dependency chain:', dependencyChain);
        console.log('Original snapshots:', originalSnapshots);
        console.log('Legitimate snapshots:', legitimateSnapshots);
        console.log('Active observers:', Object.keys(activeObservers));
        console.log('Legitimate updates pending:', Array.from(legitimateUpdates));
        console.log('Pending legitimate update timeouts:', Array.from(pendingLegitimateUpdates.keys())); */
        
        dropdownIds.forEach(id => {
            const el = document.getElementById(id);
            console.log(`${id}:`, {
                exists: !!el,
                type: el ? el.tagName : 'N/A',
                options: el ? el.options.length : 0,
                hasObserver: !!activeObservers[id],
                hasOriginalSnapshot: !!originalSnapshots[id],
                hasLegitimateSnapshot: !!legitimateSnapshots[id],
                isDependent: Object.values(dependencyChain).some(deps => deps.includes(id)),
                isParent: !!dependencyChain[id]
            });
        });
    };
    
    // Test tampering function (for debugging)
    window.testTampering = function(id, action = 'add') {
        const el = document.getElementById(id);
        if (!el) {
            console.log(`Element ${id} not found`);
            return;
        }
        
        if (action === 'add') {
            const option = document.createElement('option');
            option.value = 'test_' + Date.now();
            option.text = 'Test Option ' + Date.now();
            el.appendChild(option);
            console.log(`Added test option to ${id}`);
        } else if (action === 'remove' && el.options.length > 0) {
            el.removeChild(el.options[el.options.length - 1]);
            console.log(`Removed last option from ${id}`);
        }
    };
    
    // Start initialization
    initialize();
    
   // console.log('Dropdown Monitor Script setup complete with fixed tampering detection');
